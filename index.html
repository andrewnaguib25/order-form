let cart = []; function addToCart(name, price) { let item = cart.find(i => i.name === name); if (item) { item.quantity++; } else { cart.push({ name, price, quantity: 1 }); } updateCart(); } function updateCart() { let cartBody = $("#cartBody").empty(); let total = 0; cart.forEach((item, index) => { let itemTotal = item.quantity * item.price; total += itemTotal; cartBody.append(` ${item.name} ${item.quantity} $${itemTotal.toFixed(2)} `); }); $("#totalPrice").text(total.toFixed(2)); } function removeFromCart(index) { cart.splice(index, 1); updateCart(); } function submitOrder() { if (cart.length === 0) { alert("Your cart is empty!"); return; } alert("Order Submitted!"); cart = []; updateCart(); }
